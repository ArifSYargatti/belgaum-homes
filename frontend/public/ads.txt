export const trackEvent = (eventName, eventData = {}) => {
  if (typeof fbq !== 'undefined') {
    fbq('track', eventName, eventData);
    console.log(`Tracked ${eventName}:`, eventData);
  }
};

export const trackLead = (leadData) => {
  trackEvent('Lead', {
    content_name: leadData.apartmentTitle || 'General Inquiry',
    content_category: 'Apartment Inquiry',
    lead_type: 'Property Inquiry'
  });
};

export const trackViewContent = (apartment) => {
  trackEvent('ViewContent', {
    content_ids: [apartment.id],
    content_name: apartment.title,
    content_type: 'product',
    value: parseInt(apartment.price.replace(/[^0-9]/g, '')),
    currency: 'INR'
  });
};
